# 1. NVM 설치 스크립트 실행 (공식 GitHub 리포지토리 참조)
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.7/install.sh | bash

# 2. 터미널 프로파일 적용 (또는 터미널 재시작)
source ~/.bashrc

# 3. 최신 LTS(장기 지원) 버전의 Node.js 설치
nvm install --lts

# 4. Gemini CLI 패키지 전역(Global) 설치 (예시)
npm install -g @google/generative-ai-cli
